....

Example syntax for including the silver theme in a bundle:

\{% comment %}

[source,js]
----
import 'tinymce/themes/silver';
----

[source,js]
----
import '../tinymce/themes/silver/theme';
----

[source,js]
----
require('tinymce/themes/silver');
----

[source,js]
----
require('../tinymce/themes/silver/theme.js');
----

\{% endcomment %}
