The following table shows examples of the syntax used to bundle the following example localization file:

....
./langs/sv_SE.js
....

\{% comment %}

[source,js]
----
import 'tinymce/langs/sv_SE';
----

[source,js]
----
import '../tinymce/langs/sv_SE';
----

[source,js]
----
require('tinymce/langs/sv_SE');
----

[source,js]
----
require('../tinymce/langs/sv_SE.js');
----

\{% endcomment %}
